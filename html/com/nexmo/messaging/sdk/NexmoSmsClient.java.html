32/**
33 * NexmoSmsClient.java<br><br>
34 *
35 * Client for talking to the Nexmo REST interface<br><br>
36 *
37 * Usage
38 *
39 * To submit a message, first you should instanciate a NexmoSmsClient, passing the credentials for your Nexmo account on the constructor.
40 * Then, you should instanciate the appropriate {@link com.nexmo.messaging.sdk.messages.Message} subclass depending on which type of message you are going to submit.<br>
41 * The following subclasses are available ...<br><br>
42 *
43 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.TextMessage}<br>
44 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.BinaryMessage}<br>
45 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.WapPushMessage}<br>
46 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.UnicodeMessage}<br>
47 *
48 * Each of these subclasses requires different message parameters to be passed on the constructor.
49 * See the included javadocs for further details.<br><br>
50 *
51 * Once you have a {@link com.nexmo.messaging.sdk.messages.Message} object, you simply pass this to the {@link #submitMessage(Message)} method in the NexmoSmsClient instance.
52 * This will construct and post the request to the Nexmo REST service.<br>
53 * This method will return an array of {@link com.nexmo.messaging.sdk.SmsSubmissionResult}, with 1 entry for every sms message that was sent.<br>
54 * Certain messages, for example, long text messages greater than 160 characters, will require multiple sms's to be submitted.<br>
55 * Each entry in this array will contain an individual messageId as well as an individual status detailing the success or reason for failure of each message.<br><br>
56 *
57 * The list of possible status codes is listed below ..<br><br>
58 *
59 * // 0 = success  -- the message was successfully accepted for delivery by nexmo<br>
60 * // 1 = throttled -- You have exceeded the submission capacity allowed on this account, please back-off and re-try<br>
61 * // 2 = missing params -- Your request is incomplete and missing some mandatory parameters<br>
62 * // 3 = invalid params -- The value of 1 or more parameters is invalid<br>
63 * // 4 = invalid credentials -- The username / password you supplied is either invalid or disabled<br>
64 * // 5 = internal error -- An error has occurred in the nexmo platform whilst processing this message<br>
65 * // 6 = invalid message -- The Nexmo platform was unable to process this message, for example, an un-recognized number prefix<br>
66 * // 7 = number barred -- The number you are trying to submit to is blacklisted and may not receive messages<br>
67 * // 8 = partner account barred -- The username you supplied is for an account that has been barred from submitting messages<br>
68 * // 9 = partner quota exceeded -- Your pre-pay account does not have sufficient credit to process this message<br>
69 * // 10 = too many existing binds -- The number of simultaneous connections to the platform exceeds the capabilities of your account<br>
70 * // 11 = account not enabled for http -- This account is not provisioned for http / rest submission, you should use SMPP instead<br>
71 * // 12 = message too long -- The message length exceeds the maximum allowed<br>
72 *
73 * @author  Paul Cook
74 * @version 1.0
75 */

41    /**
42     * Instanciate a new NexmoSmsClient instance that will communicate using the supplied credentials, and will use the supplied connection and read timeout values.<br>
43     * Additionally, you can specify an alternative service base url. For example submitting to a testing sandbox environment,
44     * or if requested to submit to an alternative address by Nexmo, for example, in cases where it may be necessary to prioritize your traffic.
45     *
46     * @param username Your Nexmo account id
47     * @param password Your Nexmo account password
48     * @param connectionTimeout over-ride the default connection timeout with this value (in milliseconds)
49     * @param soTimeout over-ride the default read-timeout with this value (in milliseconds)
50     */
51    public NexmoSmsClient(String baseUrl,
52                          final String username,
53                          final String password,
54                          final int connectionTimeout,
55                          final int soTimeout) throws Exception {
56
57        // Derive a http and a https version of the supplied base url
58        baseUrl = baseUrl.trim();
59        String lc = baseUrl.toLowerCase();
60        if (!lc.startsWith("http://") && !lc.startsWith("https://"))
61            throw new Exception("base url does not start with http:// or https://");
62        if (lc.startsWith("http://")) {
63            this.baseUrlHttp = baseUrl;
64            this.baseUrlHttps = "https://" + baseUrl.substring(7);
65        } else {
66            this.baseUrlHttps = baseUrl;
67            this.baseUrlHttp = "http://" + baseUrl.substring(8);
68        }
69
70        this.username = username;
71        this.password = password;
72        this.connectionTimeout = connectionTimeout;
73        this.soTimeout = soTimeout;
74        try {
75            this.documentBuilderFactory = DocumentBuilderFactory.newInstance();
76            this.documentBuilder = this.documentBuilderFactory.newDocumentBuilder();
77        } catch (javax.xml.parsers.ParserConfigurationException e) {
78            throw new Exception("ERROR initializing XML Document builder!", e);
79        }
80    }

27     * This method will respond with an array of SmsSubmissionResult objects. Depending on the nature and length of the submitted message, Nexmo may automatically
28     * split the message into multiple sms messages in order to deliver to the handset. For example, a long text sms of greater than 160 chars will need to be split
29     * into multiple 'concatenated' sms messages. The Nexmo service will handle this automatically for you.<br>
30     * The array of SmsSubmissionResult objects will contain a SmsSubmissionResult object for every actual sms that was required to submit the message.
31     * each message can potentially have a different status result, and each message will have a different message id.
32     * Delivery notifications will be generated for each sms message within this set and will be posted to your application containing the appropriate message id.
33     *
34     * @param message The message request object that describes the type of message and the contents to be submitted.
35     * @param https Flag to indicate whether the request should be submitted using SSL/HTTPS rather than HTTP
36     *
37     * @return SmsSubmissionResult[] an array of results, 1 object for each sms message that was required to submit this message in its entirety
38     *
39     * @throws Exception There has been a general failure either within the Client class, or whilst attempting to communicate with the Nexmo service (eg, Network failure)
40     */
41    public SmsSubmissionResult[] submitMessage(Message message, boolean https) throws Exception {
42
43        log.debug("HTTP-SMS-Submission Client .. from [ " + message.getFrom() + " ] to [ " + message.getTo() + " ] msg [ " + message.getMessageBody() + " ] ");
44
45        // From the Message object supplied, construct an appropriate request to be submitted to the Nexmo REST Service.
46
47        // Determine the type parameter based on the type of Message object.
48        boolean binary = message.getType() == Message.MESSAGE_TYPE_BINARY;
49        boolean unicode = message.isUnicode();
50        boolean wapPush = message.getType() == Message.MESSAGE_TYPE_WAPPUSH;
51
52        String mode = "text";
53
54        if (binary)
55            mode = "binary";
56
57        if (unicode)
58            mode = "unicode";
59
60        if (wapPush)
61            mode = "wappush";
62
63        // Construct a query string as a list of NameValuePairs
64
65        List<NameValuePair> params = new ArrayList();
66        boolean doPost = false;
67
68        params.add(new NameValuePair("username", this.username));
69        params.add(new NameValuePair("password", this.password));
70        params.add(new NameValuePair("from", message.getFrom()));
71        params.add(new NameValuePair("to", message.getTo()));
72        params.add(new NameValuePair("type", mode));
73        if (wapPush) {
74            params.add(new NameValuePair("url", message.getWapPushUrl()));
75            params.add(new NameValuePair("title", message.getWapPushTitle()));
76            if (message.getWapPushValidity() > 0)
77                params.add(new NameValuePair("validity", "" + message.getWapPushValidity()));
78        } else if (binary) {
79            // Binary Message
80            if (message.getBinaryMessageUdh() != null)
81                params.add(new NameValuePair("udh", HexUtil.bytesToHex(message.getBinaryMessageUdh())));
82            params.add(new NameValuePair("body", HexUtil.bytesToHex(message.getBinaryMessageBody())));
83        } else {
84            // Text Message
85            params.add(new NameValuePair("text", message.getMessageBody()));
86            if (message.getMessageBody() != null && message.getMessageBody().length() > 255)
87                doPost = true;
88        }
89
90        if (message.getClientReference() != null)
91            params.add(new NameValuePair("client-ref", message.getClientReference()));
92
93        params.add(new NameValuePair("status-report-req", "" + message.getStatusReportRequired()));
94
95        NameValuePair[] queryString = (NameValuePair[])params.toArray(new NameValuePair[params.size()]);
96
97        String baseUrl = https ? this.baseUrlHttps : this.baseUrlHttp;
98
99        // Now that we have generated a query string, we can instanciate a HttpClient,
00        // construct a POST or GET method and execute to submit the request
01        String response = null;
02        for (int pass=1;pass<=2;pass++) {
03            HttpMethod method = null;
04            if (doPost)
05                method = new PostMethod(baseUrl);
06            else
07                method = new GetMethod(baseUrl);
08            method.setQueryString(EncodingUtil.formUrlEncode(queryString, "UTF-8"));
09            String url = baseUrl + "?" + method.getQueryString();
10            try {
11                if (this.httpClient == null)
12                    this.httpClient = HttpClientUtils.getInstance(this.connectionTimeout, this.soTimeout).getNewHttpClient();
13                int status = this.httpClient.executeMethod(method);
14                if (status != 200)
15                    throw new Exception("got a non-200 response [ " + status + " ] from Nexmo-HTTP for url [ " + url + " ] ");
16                response = method.getResponseBodyAsString();
17                method.releaseConnection();
18                log.info(".. SUBMITTED NEXMO-HTTP URL [ " + url + " ] -- response [ " + response + " ] ");
19                break;
20            } catch (Exception e) {
21                if (method != null)
22                    method.releaseConnection();
23                log.info("communication failure: " + e);
24                String exceptionMsg = e.getMessage();
25                if (exceptionMsg.indexOf("Read timed out") >= 0) {
26                    log.info("we're still connected, but the target did not respond in a timely manner ..  drop ...");
27                } else {
28                    if (pass == 1) {
29                        log.info("... re-establish http client ...");
30                        this.httpClient = null;
31                        continue;
32                    }
33                }
34
35                // return a COMMS failure ...
36                SmsSubmissionResult[] results = new SmsSubmissionResult[1];
37                results[0] = new SmsSubmissionResult(SmsSubmissionResult.STATUS_COMMS_FAILURE,
38                                                     null,
39                                                     "Failed to communicate with NEXMO-HTTP url [ " + url + " ] ..." + e,
40                                                     message.getClientReference(),
41                                                     null,
42                                                     null,
43                                                     true);
44                return results;
45            }
46        }
47
48        // parse the response doc ...
49
50        /*
51            We receive a response from the api that looks like this, parse the document
52            and turn it into an array of SmsSubmissionResult, one object per <message> node
53
54                <mt-submission-response>
55                    <messages count='x'>
56                        <message>
57                            <messageId>xxx</messageId>
58                            <status>xx</status>
59                            <errorText>ff</errorText>
60                            <clientRef>xxx</clientRef>
61                            <remainingBalance>##.##</remainingBalance>
62                            <messagePrice>##.##</messagePrice>
63                        </message>
64                    </messages>
65                </mt-submission-response>
66
67            Response codes ...
68
69                // 0 = success
70                // 1 = throttled
71                // 2 = missing params
72                // 3 = invalid params
73                // 4 = invalid credentials
74                // 5 = internal error
75                // 6 = invalid message
76                // 7 = number barred
77                // 8 = partner account barred
78                // 9 = partner quota exceeded
79                // 10 = too many existing binds
80                // 11 = account not enabled for http
81                // 12 = message too long
82
83        */
84
85        List<SmsSubmissionResult> results = new ArrayList();
86
87        Document doc = null;
88        synchronized(this.documentBuilder) {
89            try {
90                doc = this.documentBuilder.parse(new InputSource(new StringReader(response)));
91            } catch (Exception e) {
92                throw new Exception("Failed to build a DOM doc for the xml document [ " + response + " ] ", e);
93            }
94        }
95
96        NodeList replies = doc.getElementsByTagName("mt-submission-response");
97        for (int i=0;i<replies.getLength();i++) {
98            Node reply = replies.item(i);
99            NodeList messageLists = reply.getChildNodes();
00            for (int i2=0;i2<messageLists.getLength();i2++) {
01                Node messagesNode = messageLists.item(i2);
02                if (messagesNode.getNodeType() != Node.ELEMENT_NODE)
03                    continue;
04                if (messagesNode.getNodeName().equals("messages")) {
05                    NodeList messages = messagesNode.getChildNodes();
06                    for (int i3=0;i3<messages.getLength();i3++) {
07                        Node messageNode = messages.item(i3);
08                        if (messageNode.getNodeType() != Node.ELEMENT_NODE)
09                            continue;
10
11                        int status = -1;
12                        String messageId = null;
13                        String errorText = null;
14                        String clientReference = null;
15                        BigDecimal remainingBalance = null;
16                        BigDecimal messagePrice = null;
17
18                        NodeList nodes = messageNode.getChildNodes();
19                        for (int i4=0;i4<nodes.getLength();i4++) {
20                            Node node = nodes.item(i4);
21                            if (node.getNodeType() != Node.ELEMENT_NODE)
22                                continue;
23                            if (node.getNodeName().equals("messageId")) {
24                                messageId = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
25                            } else if (node.getNodeName().equals("status")) {
26                                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
27                                try {
28                                    status = Integer.parseInt(str);
29                                } catch (Exception e) {
30                                    log.error("xml parser .. invalid value in <status> node [ " + str + " ] ");
31                                    status = SmsSubmissionResult.STATUS_INTERNAL_ERROR;
32                                }
33                            } else if (node.getNodeName().equals("errorText")) {
34                                errorText = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
35                            } else if (node.getNodeName().equals("clientRef")) {
36                                clientReference = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
37                            } else if (node.getNodeName().equals("remainingBalance")) {
38                                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
39                                try {
40                                    remainingBalance = new BigDecimal(str);
41                                } catch (Exception e) {
42                                    log.error("xml parser .. invalid value in <remainingBalance> node [ " + str + " ] ");
43                                }
44                            } else if (node.getNodeName().equals("messagePrice")) {
45                                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
46                                try {
47                                    messagePrice = new BigDecimal(str);
48                                } catch (Exception e) {
49                                    log.error("xml parser .. invalid value in <messagePrice> node [ " + str + " ] ");
50                                }
51                            } else
52                                log.error("xml parser .. unknown node found in status-return, should be [ message-id, status, error-text or client-ref value ] -- found [ " + node.getNodeName() + " ] ");
53                        }
54
55                        if (status == -1)
56                            throw new Exception("Xml Parser - did not find a <status> node");
57
58                        // Is this a temporary error ?
59                        boolean temporaryError = (status == SmsSubmissionResult.STATUS_THROTTLED ||
60                                                  status == SmsSubmissionResult.STATUS_INTERNAL_ERROR ||
61                                                  status == SmsSubmissionResult.STATUS_TOO_MANY_BINDS);
62
63                        results.add(new SmsSubmissionResult(status,
64                                                            messageId,
65                                                            errorText,
66                                                            clientReference,
67                                                            remainingBalance,
68                                                            messagePrice,
69                                                            temporaryError));
70                    }
71                }
72            }
73        }
74
75        return results.toArray(new SmsSubmissionResult[0]);
76    }
