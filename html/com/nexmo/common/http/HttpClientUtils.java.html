9 /**

18 */
19public class HttpClientUtils {
20
21    private final static Map<String, HttpClientUtils> instances = new HashMap<String, HttpClientUtils>();
22
23    private final MultiThreadedHttpConnectionManager multiThreadedHttpConnectionManager;
24
25    private HttpClientUtils(int connectionTimeout, int soTimeout) {
26        this.multiThreadedHttpConnectionManager = new MultiThreadedHttpConnectionManager();
27        this.multiThreadedHttpConnectionManager.getParams().setConnectionTimeout(connectionTimeout);
28        this.multiThreadedHttpConnectionManager.getParams().setSoTimeout(soTimeout);
29        this.multiThreadedHttpConnectionManager.getParams().setMaxTotalConnections(200);
30        this.multiThreadedHttpConnectionManager.getParams().setDefaultMaxConnectionsPerHost(200);
31        this.multiThreadedHttpConnectionManager.getParams().setTcpNoDelay(true);
32    }
33
34    /**
35     * Return an existing or instanciate a new HttpClient factory instance with explicitely specified connection and read timeout values
36     *
37     * @param connectionTimeout the timeout value in milliseconds to use when establishing a new http socket
38     * @param soTimeout the timeout value in milliseconds to wait for a http response before closing the socket
39     *
40     * @return HttpClientUtils an instance of the HttpClient factory primed with the requested timeout values
41     */
42    public static HttpClientUtils getInstance(int connectionTimeout, int soTimeout) {
43        String key = "c-" + connectionTimeout + "-so-" + soTimeout;
44        HttpClientUtils instance = instances.get(key);
45        if (instance == null) {
46            instance = new HttpClientUtils(connectionTimeout, soTimeout);
47            instances.put(key, instance);
48        }
49        return instance;
50    }
51
52    /**
53     * Instanciate a new HttpClient instance that uses the timeout values associated with this factory instance
54     *
55     * @return HttpClient a new HttpClient instance
56     */
57    public HttpClient getNewHttpClient() {
58        return new HttpClient(this.multiThreadedHttpConnectionManager);
59    }
60
61}
