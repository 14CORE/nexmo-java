022package com.nexmo.client.voice;
023
024
025import com.nexmo.client.AbstractClient;
026import com.nexmo.client.HttpWrapper;
027import com.nexmo.client.NexmoClient;
028import com.nexmo.client.NexmoClientException;
029import com.nexmo.client.voice.endpoints.CallsEndpoint;
030import com.nexmo.client.voice.endpoints.DtmfEndpoint;
031import com.nexmo.client.voice.endpoints.StreamsEndpoint;
032import com.nexmo.client.voice.endpoints.TalkEndpoint;
033
034import java.io.IOException;
035
036/**
037 * A client for talking to the Nexmo Voice API. The standard way to obtain an instance of this class is to use
038 * {@link NexmoClient#getVoiceClient()}.
039 */
040public class VoiceClient extends AbstractClient {
041    protected final CallsEndpoint calls;
042    protected final StreamsEndpoint streams;
043    protected final TalkEndpoint talk;
044    protected final DtmfEndpoint dtmf;
045
046    /**
047     * Constructor.
048     *
049     * @param httpWrapper (required) shared HTTP wrapper object used for making REST calls.
050     */
051    public VoiceClient(HttpWrapper httpWrapper) {
052        super(httpWrapper);
053
054        calls = new CallsEndpoint(httpWrapper);
055        streams = new StreamsEndpoint(httpWrapper);
056        talk = new TalkEndpoint(httpWrapper);
057        dtmf = new DtmfEndpoint(httpWrapper);
058    }
059
060    /**
061     * Begin a call to a phone number.
062     *
063     * @param callRequest Describing the call to be made.
064     * @return A CallEvent describing the initial state of the call, containing the <tt>uuid</tt> required to
065     * interact with the ongoing phone call.
066     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
067     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
068     */
069    public CallEvent createCall(Call callRequest) throws IOException, NexmoClientException {
070        return calls.post(callRequest);
071    }
072
073    /**
074     * Obtain the first page of CallInfo objects, representing the most recent calls initiated by
075     * {@link #createCall(Call)}.
076     *
077     * @return A CallInfoPage representing the response from the Nexmo Voice API.
078     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
079     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
080     */
081    public CallInfoPage listCalls() throws IOException, NexmoClientException {
082        return this.listCalls(null);
083    }
084
085    /**
086     * Obtain the first page of CallInfo objects matching the query described by <tt>filter</tt>, representing the most
087     * recent calls initiated by {@link #createCall(Call)}.
088     *
089     * @param filter (optional) A filter describing which calls to be listed.
090     * @return A CallInfoPage representing the response from the Nexmo Voice API.
091     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
092     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
093     */
094    public CallInfoPage listCalls(CallsFilter filter) throws IOException, NexmoClientException {
095        return calls.get(filter);
096    }
097
098    /**
099     * Look up the status of a single call initiated by {@link #createCall(Call)}.
100     *
101     * @param uuid (required) The UUID of the call, obtained from the object returned by {@link #createCall(Call)}.
102     *             This value can be obtained with {@link CallEvent#getUuid()}
103     * @return A CallInfo object, representing the response from the Nexmo Voice API.
104     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
105     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
106     */
107    public CallInfo getCallDetails(String uuid) throws IOException, NexmoClientException {
108        return calls.get(uuid);
109    }
110
111    /**
112     * Send DTMF codes to an ongoing call.
113     *
114     * @param uuid   (required) The UUID of the call, obtained from the object returned by {@link #createCall(Call)}.
115     *               This value can be obtained with {@link CallEvent#getUuid()}
116     * @param digits (required) A string specifying the digits to be sent to the call. Valid characters are the digits
117     *               <tt>1-9</tt>, <tt>#</tt>, <tt>*</tt>, with the special character <tt>p</tt> indicating a short
118     *               pause between tones.
119     * @return A CallInfo object, representing the response from the Nexmo Voice API.
120     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
121     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
122     */
123    public DtmfResponse sendDtmf(String uuid, String digits) throws IOException, NexmoClientException {
124        return dtmf.put(uuid, digits);
125    }
126
127    /**
128     * Hang up a call.
129     * <p>
130     * In future, further operations will be possible. At the moment, the only valid value for <tt>action</tt> is
131     * hangup.
132     *
133     * @param uuid   (required) The UUID of the call, obtained from the object returned by {@link #createCall(Call)}.
134     *               This value can be obtained with {@link CallEvent#getUuid()}
135     * @param action The word "hangup"
136     * @return A CallInfo object, representing the response from the Nexmo Voice API.
137     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
138     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
139     */
140    public ModifyCallResponse modifyCall(String uuid, String action) throws IOException, NexmoClientException {
141        return calls.put(uuid, action);
142    }
143
144    /**
145     * Stream audio to an ongoing call.
146     *
147     * @param uuid      The UUID of the call, obtained from the object returned by {@link #createCall(Call)}. This value can
148     *                  be obtained with {@link CallEvent#getUuid()}
149     * @param streamUrl A URL of an audio file in MP3 or 16-bit WAV format, to be streamed to the call.
150     * @param loop      The number of times to repeat the audio. The default value is <tt>1</tt>, or you can use
151     *                  <tt>0</tt> to indicate that the audio should be repeated indefinitely.
152     * @return The data returned from the Voice API
153     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
154     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
155     */
156    public StreamResponse startStream(String uuid, String streamUrl, int loop) throws IOException, NexmoClientException {
157        return streams.put(new StreamRequest(uuid, streamUrl, loop));
158    }
159
160    /**
161     * Stream audio to an ongoing call.
162     *
163     * @param uuid      The UUID of the call, obtained from the object returned by {@link #createCall(Call)}. This value can
164     *                  be obtained with {@link CallEvent#getUuid()}
165     * @param streamUrl A URL of an audio file in MP3 or 16-bit WAV format, to be streamed to the call.
166     * @return The data returned from the Voice API.
167     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
168     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
169     */
170    public StreamResponse startStream(String uuid, String streamUrl) throws IOException, NexmoClientException {
171        return streams.put(new StreamRequest(uuid, streamUrl, 1));
172    }
173
174    /**
175     * Stop the audio being streamed into a call.
176     *
177     * @param uuid The UUID of the call, obtained from the object returned by {@link #createCall(Call)}. This value can
178     *             be obtained with {@link CallEvent#getUuid()}
179     * @return The data returned from the Voice API
180     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
181     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
182     */
183    public StreamResponse stopStream(String uuid) throws IOException, NexmoClientException {
184        return streams.delete(uuid);
185    }
186
187    /**
188     * Send a synthesized speech message to an ongoing call.
189     * <p>
190     * The message will only play once, spoken with the default voice of Kimberly.
191     *
192     * @param uuid The UUID of the call, obtained from the object returned by {@link #createCall(Call)}. This value can
193     *             be obtained with {@link CallEvent#getUuid()}
194     * @param text The message to be spoken to the call participants.
195     * @return The data returned from the Voice API.
196     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
197     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
198     */
199    public TalkResponse startTalk(String uuid, String text) throws IOException, NexmoClientException {
200        return talk.put(new TalkRequest(uuid, text));
201    }
202
203    /**
204     * Send a synthesized speech message to an ongoing call.
205     *
206     * @param uuid      The UUID of the call, obtained from the object returned by {@link #createCall(Call)}. This value can
207     *                  be obtained with {@link CallEvent#getUuid()}
208     * @param text      The message to be spoken to the call participants.
209     * @param voiceName The voice to be used to speak the message.
210     * @return The data returned from the Voice API.
211     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
212     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
213     */
214    public TalkResponse startTalk(String uuid, String text, VoiceName voiceName) throws IOException, NexmoClientException {
215        return talk.put(new TalkRequest(uuid, text, voiceName));
216    }
217
218    /**
219     * Send a synthesized speech message to an ongoing call.
220     * <p>
221     * The message will be spoken with the default voice of Kimberly.
222     *
223     * @param uuid The UUID of the call, obtained from the object returned by {@link #createCall(Call)}. This value can
224     *             be obtained with {@link CallEvent#getUuid()}
225     * @param text The message to be spoken to the call participants.
226     * @param loop The number of times to repeat the message. The default value is <tt>1</tt>, or you can use
227     *             <tt>0</tt> to indicate that the message should be repeated indefinitely.
228     * @return The data returned from the Voice API.
229     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
230     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
231     */
232    public TalkResponse startTalk(String uuid, String text, int loop) throws IOException, NexmoClientException {
233        return talk.put(new TalkRequest(uuid, text, loop));
234    }
235
236    /**
237     * Send a synthesized speech message to an ongoing call.
238     *
239     * @param uuid      The UUID of the call, obtained from the object returned by {@link #createCall(Call)}. This value can
240     *                  be obtained with {@link CallEvent#getUuid()}
241     * @param text      The message to be spoken to the call participants.
242     * @param voiceName The voice to be used to speak the message.
243     * @param loop      The number of times to repeat the message. The default value is <tt>1</tt>, or you can use
244     *                  <tt>0</tt> to indicate that the message should be repeated indefinitely.
245     * @return The data returned from the Voice API.
246     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
247     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
248     */
249    public TalkResponse startTalk(String uuid, String text, VoiceName voiceName, int loop) throws IOException, NexmoClientException {
250        return talk.put(new TalkRequest(uuid, text, voiceName, loop));
251    }
252
253    /**
254     * Stop the message being spoken into a call.
255     *
256     * @param uuid The UUID of the call, obtained from the object returned by {@link #createCall(Call)}. This value can
257     *             be obtained with {@link CallEvent#getUuid()}
258     * @return The data returned from the Voice API
259     * @throws IOException          if a network error occurred contacting the Nexmo Voice API.
260     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
261     */
262    public TalkResponse stopTalk(String uuid) throws IOException, NexmoClientException {
263        return talk.delete(uuid);
264    }
