022package com.nexmo.client.verify.endpoints;
023
024import com.nexmo.client.HttpWrapper;
025import com.nexmo.client.NexmoClientException;
026import com.nexmo.client.NexmoResponseParseException;
027import com.nexmo.client.auth.TokenAuthMethod;
028import com.nexmo.client.legacyutils.XmlParser;
029import com.nexmo.client.legacyutils.XmlUtil;
030import com.nexmo.client.verify.BaseResult;
031import com.nexmo.client.verify.CheckRequest;
032import com.nexmo.client.verify.CheckResult;
033import com.nexmo.client.voice.endpoints.AbstractMethod;
034import org.apache.commons.logging.Log;
035import org.apache.commons.logging.LogFactory;
036import org.apache.http.HttpResponse;
037import org.apache.http.client.methods.RequestBuilder;
038import org.apache.http.impl.client.BasicResponseHandler;
039import org.w3c.dom.Document;
040import org.w3c.dom.Element;
041import org.w3c.dom.Node;
042import org.w3c.dom.NodeList;
043
044import java.io.IOException;
045import java.io.UnsupportedEncodingException;
046
047public class VerifyCheckMethod extends AbstractMethod<CheckRequest, CheckResult> {
048    private static final Log log = LogFactory.getLog(VerifyCheckMethod.class);
049    private static final Class[] ALLOWED_AUTH_METHODS = new Class[]{TokenAuthMethod.class};
050
051    private static final String DEFAULT_URI = "https://api.nexmo.com/verify/check/xml";
052
053    private XmlParser xmlParser = new XmlParser();
054    private String uri = DEFAULT_URI;
055
056    public VerifyCheckMethod(HttpWrapper httpWrapper) {
057        super(httpWrapper);
058    }
059
060    @Override
061    protected Class[] getAcceptableAuthMethods() {
062        return ALLOWED_AUTH_METHODS;
063    }
064
065    @Override
066    public RequestBuilder makeRequest(CheckRequest request) throws NexmoClientException, UnsupportedEncodingException {
067        if (request.getRequestId() == null || request.getCode() == null)
068            throw new IllegalArgumentException("request ID and code parameters are mandatory.");
069
070        RequestBuilder result = RequestBuilder.post(this.uri)
071                .addParameter("request_id", request.getRequestId())
072
073                .addParameter("code", request.getCode());
074        if (request.getIpAddress() != null)
075            result.addParameter("ip_address", request.getIpAddress());
076
077        return result;
078    }
079
080    @Override
081    public CheckResult parseResponse(HttpResponse response) throws IOException {
082        String body = new BasicResponseHandler().handleResponse(response);
083        return parseCheckResponse(body);
084    }
085
086    private CheckResult parseCheckResponse(String response) throws NexmoResponseParseException {
087        Document doc = xmlParser.parseXml(response);
088
089        Element root = doc.getDocumentElement();
090        if (!"verify_response".equals(root.getNodeName()))
091            throw new NexmoResponseParseException("No valid response found [ " + response + "] ");
092
093        String eventId = null;
094        int status = -1;
095        float price = -1;
096        String currency = null;
097        String errorText = null;
098
099        NodeList fields = root.getChildNodes();
100        for (int i = 0; i < fields.getLength(); i++) {
101            Node node = fields.item(i);
102            if (node.getNodeType() != Node.ELEMENT_NODE)
103                continue;
104
105            String name = node.getNodeName();
106            if ("event_id".equals(name)) {
107                eventId = XmlUtil.stringValue(node);
108            } else if ("status".equals(name)) {
109                String str = XmlUtil.stringValue(node);
110                try {
111                    if (str != null)
112                        status = Integer.parseInt(str);
113                } catch (NumberFormatException e) {
114                    log.error("xml parser .. invalid value in <status> node [ " + str + " ] ");
115                    status = BaseResult.STATUS_INTERNAL_ERROR;
116                }
117            } else if ("price".equals(name)) {
118                String str = XmlUtil.stringValue(node);
119                try {
120                    if (str != null)
121                        price = Float.parseFloat(str);
122                } catch (NumberFormatException e) {
123                    log.error("xml parser .. invalid value in <price> node [ " + str + " ] ");
124                }
125            } else if ("currency".equals(name)) {
126                currency = XmlUtil.stringValue(node);
127            } else if ("error_text".equals(name)) {
128                errorText = XmlUtil.stringValue(node);
129            }
130        }
131
132        if (status == -1)
133            throw new NexmoResponseParseException("Xml Parser - did not find a <status> node");
134
135        // Is this a temporary error ?
136        boolean temporaryError = (status == BaseResult.STATUS_THROTTLED || status == BaseResult.STATUS_INTERNAL_ERROR);
137
138        return new CheckResult(status, eventId, price, currency, errorText, temporaryError);
139    }
140}
