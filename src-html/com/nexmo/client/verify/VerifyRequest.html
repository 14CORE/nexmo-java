022package com.nexmo.client.verify;
023
024import java.util.Locale;
025
026/**
027 * Describes a Verify request when passed to
028 * {@link com.nexmo.client.verify.endpoints.VerifyEndpoint#execute(Object)})}.
029 * <p>
030 * <b>Note</b>: This is currently an internal object. Use {@link VerifyClient#verify} methods instead of using this
031 * class directly.
032 */
033public class VerifyRequest {
034    // TODO: This class does not represent all values supported by the verification API.
035    private final String number;
036    private final String brand;
037    private final String from;
038    private final int length;
039    private final Locale locale;
040    private final LineType type;
041
042    /**
043     * Constructor.
044     *
045     * @param number (required) The recipient's phone number in <a href="https://en.wikipedia.org/wiki/E.164">E.164</a>
046     *               format.
047     * @param brand  (required) The name of the company or app to be verified for. Must not be longer than 18
048     *               characters.
049     */
050    public VerifyRequest(final String number, final String brand) {
051        this(number,
052                brand,
053                null,
054                -1,
055                null,
056                null);
057    }
058
059    /**
060     * Constructor.
061     *
062     * @param number (required) The recipient's phone number in <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
063     * @param brand  (required) The name of the company or app you are verifying for. Must not be longer than 18
064     *               characters.
065     * @param from   (optional The Nexmo number to use as the sender for the verification SMS message and calls, in
066     *               <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
067     */
068    public VerifyRequest(final String number, final String brand, final String from) {
069        this(number, brand, from, -1, null, null);
070    }
071
072    /**
073     * Constructor.
074     *
075     * @param number (required) The recipient's phone number in <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
076     * @param brand  (required) The name of the company or app you are verifying for. Must not be longer than 18
077     *               characters.
078     * @param from   (optional The Nexmo number to use as the sender for the verification SMS message and calls, in
079     *               <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
080     * @param length (optional) The length of the verification code to be sent to the user. Must be either 4 or 6. Use
081     *               -1 to use the default value.
082     * @param locale (optional) Override the default locale used for verification. By default the locale is determined
083     *               from the country code included in <tt>number</tt>
084     */
085    public VerifyRequest(final String number, final String brand, final String from, final int length,
086                         final Locale locale) {
087        this(number, brand, from, length, locale, null);
088    }
089
090    /**
091     * Constructor.
092     *
093     * @param number (required) The recipient's phone number in <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
094     * @param brand  (required) The name of the company or app you are verifying for. Must not be longer than 18
095     *               characters.
096     * @param from   (optional A short alphanumeric string to specify the SenderID for SMS sent by Verify.
097     *               Depending on the destination of the phone number you are applying, restrictions may apply.
098     *               By default, sender_id is <tt>VERIFY</tt>. Must be 11 characters or fewer.
099     * @param length (optional) The length of the verification code to be sent to the user. Must be either 4 or 6. Use
100     *               -1 to use the default value.
101     * @param locale (optional) Override the default locale used for verification. By default the locale is determined
102     *               from the country code included in <tt>number</tt>
103     * @param type   (optional) If provided, restrict the verification to the specified network type. Contact
104     *               support@nexmo.com to enable this feature.
105     */
106    public VerifyRequest(final String number, final String brand, final String from, final int length,
107                         final Locale locale, final LineType type) {
108        if (number == null || brand == null)
109            throw new IllegalArgumentException("Number and brand parameters cannot be null.");
110        if (length > 0 && length != 4 && length != 6)
111            throw new IllegalArgumentException("Code length must be 4 or 6.");
112
113        this.number = number;
114        this.brand = brand;
115        this.from = from;
116        this.length = length;
117        this.locale = locale;
118        this.type = type;
119    }
120
121    /**
122     * @return the recipient's phone number provided in the constructor, in
123     * <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
124     */
125    public String getNumber() {
126        return number;
127    }
128
129    /**
130     * @return the name of the company or app to be verified for.
131     */
132    public String getBrand() {
133        return brand;
134    }
135
136    /**
137     * @return the short alphanumeric string to specify the SenderID for SMS sent by Verify, or <tt>null</tt> if one was
138     * not provided. This value is specified in some {@link VerifyRequest} constructors.
139     * <p>
140     * If this value is <tt>null</tt>, the sender_id used will be <tt>VERIFY</tt>.
141     */
142    public String getFrom() {
143        return from;
144    }
145
146    /**
147     * @return the length of the verification code to be sent to the user, specified in some {@link VerifyRequest}
148     * constructors. <tt>-1</tt> indicates the default length will be used.
149     */
150    public int getLength() {
151        return length;
152    }
153
154    /**
155     * @return the default locale used for verification. If this value is <tt>null</tt>, the locale will be determined
156     * from the country code included in <tt>number</tt>
157     */
158    public Locale getLocale() {
159        return locale;
160    }
161
162    /**
163     * @return the type of network the verification will be restricted to. This value has no effect unless it has been
164     * enabled by contacting <tt>support@nexmo.com</tt>.
165     */
166    public LineType getType() {
167        return type;
168    }
169
170    /**
171     * Types of phone line to be specified for {@link VerifyRequest#type}.
172     */
173    public enum LineType {
174        ALL,
175        MOBILE,
176        LANDLINE,
177    }
178}
