022package com.nexmo.client.verify;
023
024
025import com.nexmo.client.AbstractClient;
026import com.nexmo.client.HttpWrapper;
027import com.nexmo.client.NexmoClient;
028import com.nexmo.client.NexmoClientException;
029import com.nexmo.client.verify.endpoints.CheckEndpoint;
030import com.nexmo.client.verify.endpoints.SearchEndpoint;
031import com.nexmo.client.verify.endpoints.VerifyEndpoint;
032
033import java.io.IOException;
034import java.util.Locale;
035
036/**
037 * A client for talking to the Nexmo Verify API. The standard way to obtain an instance of this class is to use
038 * {@link NexmoClient#getVerifyClient()}.
039 * <p>
040 * Send a verification request with a call to {@link #verify}, confirm the code entered by the user with
041 * {@link #check}, and search in-progress or completed verification requests with {@link #search}
042 * <p>
043 * More information on method parameters can be found at Nexmo website:
044 * <a href="https://docs.nexmo.com/verify">https://docs.nexmo.com/verify</a>
045 */
046public class VerifyClient extends AbstractClient {
047
048    private CheckEndpoint check;
049    private VerifyEndpoint verify;
050    private SearchEndpoint search;
051
052    /**
053     * Constructor.
054     *
055     * @param httpWrapper (required) shared HTTP wrapper object used for making REST calls.
056     */
057    public VerifyClient(HttpWrapper httpWrapper) {
058        super(httpWrapper);
059
060        this.check = new CheckEndpoint(httpWrapper);
061        this.search = new SearchEndpoint(httpWrapper);
062        this.verify = new VerifyEndpoint(httpWrapper);
063    }
064
065    /**
066     * Send a verification request to a phone number.
067     *
068     * @param number (required) The recipient's phone number in <a href="https://en.wikipedia.org/wiki/E.164">E.164</a>
069     *               format.
070     * @param brand  (required) The name of the company or app to be verified for. Must not be longer than 18
071     *               characters.
072     * @return a VerifyResult representing the response received from the Verify API call.
073     * @throws IOException          if a network error occurred contacting the Nexmo Verify API.
074     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
075     */
076    public VerifyResult verify(final String number,
077                               final String brand) throws IOException,
078                                                          NexmoClientException {
079        return verify.verify(number, brand);
080    }
081
082    /**
083     * Send a verification request to a phone number.
084     *
085     * @param number (required) The recipient's phone number in <a href="https://en.wikipedia.org/wiki/E.164">E.164</a>
086     *               format.
087     * @param brand  (required) The name of the company or app to be verified for. Must not be longer than 18
088     *               characters.
089     * @param from   (optional The Nexmo number to use as the sender for the verification SMS message and calls, in
090     *               <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
091     * @return a VerifyResult representing the response received from the Verify API call.
092     * @throws IOException          if a network error occurred contacting the Nexmo Verify API.
093     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
094     */
095    public VerifyResult verify(final String number,
096                               final String brand,
097                               final String from) throws IOException,
098                                                         NexmoClientException {
099        return verify.verify(number, brand, from);
100    }
101
102    /**
103     * Send a verification request to a phone number.
104     *
105     * @param number (required) The recipient's phone number in <a href="https://en.wikipedia.org/wiki/E.164">E.164</a>
106     *               format.
107     * @param brand  (required) The name of the company or app to be verified for. Must not be longer than 18
108     *               characters.
109     * @param from   (optional The Nexmo number to use as the sender for the verification SMS message and calls, in
110     *               <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
111     * @param length (optional) The length of the verification code to be sent to the user. Must be either 4 or 6. Use
112     *               -1 to use the default value.
113     * @param locale (optional) Override the default locale used for verification. By default the locale is determined
114     *               from the country code included in <tt>number</tt>
115     * @return a VerifyResult representing the response received from the Verify API call.
116     * @throws IOException          if a network error occurred contacting the Nexmo Verify API.
117     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
118     */
119    public VerifyResult verify(final String number,
120                               final String brand,
121                               final String from,
122                               final int length,
123                               final Locale locale) throws IOException,
124                                                           NexmoClientException {
125        return verify.verify(number, brand, from, length, locale);
126    }
127
128    /**
129     * Send a verification request to a phone number.
130     *
131     * @param number (required) The recipient's phone number in <a href="https://en.wikipedia.org/wiki/E.164">E.164</a>
132     *               format.
133     * @param brand  (required) The name of the company or app to be verified for. Must not be longer than 18
134     *               characters.
135     * @param from   (optional The Nexmo number to use as the sender for the verification SMS message and calls, in
136     *               <a href="https://en.wikipedia.org/wiki/E.164">E.164</a> format.
137     * @param length (optional) The length of the verification code to be sent to the user. Must be either 4 or 6. Use
138     *               -1 to use the default value.
139     * @param locale (optional) Override the default locale used for verification. By default the locale is determined
140     *               from the country code included in <tt>number</tt>
141     * @param type   (optional) If provided, restrict the verification to the specified network type. Contact
142     *               support@nexmo.com to enable this feature.
143     * @return a VerifyResult representing the response received from the Verify API call.
144     * @throws IOException          if a network error occurred contacting the Nexmo Verify API.
145     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
146     */
147    public VerifyResult verify(final String number,
148                               final String brand,
149                               final String from,
150                               final int length,
151                               final Locale locale,
152                               final VerifyRequest.LineType type) throws IOException,
153                                                                         NexmoClientException {
154        return verify.verify(number, brand, from, length, locale, type);
155    }
156
157    /**
158     * Validate a code provided by a user in response to a call from {@link #verify}.
159     *
160     * @param requestId (required) The requestId returned by the <tt>verify</tt> call.
161     * @param code      (required) The code entered by the user.
162     * @return a CheckResult representing the response received from the API call.
163     * @throws IOException          if a network error occurred contacting the Nexmo Verify API.
164     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
165     */
166    public CheckResult check(final String requestId,
167                             final String code) throws IOException, NexmoClientException {
168        return check.check(requestId, code, null);
169    }
170
171    /**
172     * Validate a code provided by a user in response to a call from {@link #verify}.
173     *
174     * @param requestId (required) The requestId returned by the <tt>verify</tt> call.
175     * @param code      (required) The code entered by the user.
176     * @param ipAddress (optional) The IP address obtained from the HTTP request made when the user entered their code.
177     * @return a CheckResult representing the response received from the API call.
178     * @throws IOException          if a network error occurred contacting the Nexmo Verify API.
179     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
180     */
181    public CheckResult check(final String requestId,
182                             final String code,
183                             final String ipAddress) throws IOException, NexmoClientException {
184        return check.check(requestId, code, ipAddress);
185    }
186
187    /**
188     * @param requestId The requestId of a single Verify request to be looked up.
189     * @return A SearchResult containing the details of the Verify request that was looked up, or <tt>null</tt> if no
190     * record was found.
191     * @throws IOException          if a network error occurred contacting the Nexmo Verify API.
192     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
193     */
194    public SearchResult search(String requestId) throws IOException, NexmoClientException {
195        return search.search(requestId);
196    }
197
198    /**
199     * @param requestIds The requestIds of Verify requests to be looked up.
200     * @return An array SearchResult for each record that was found.
201     * @throws IOException          if a network error occurred contacting the Nexmo Verify API.
202     * @throws NexmoClientException if there was a problem with the Nexmo request or response objects.
203     */
204    public SearchResult[] search(String... requestIds) throws IOException, NexmoClientException {
205        return search.search(requestIds);
206    }
207}
