022package com.nexmo.client;
023
024
025import com.nexmo.client.account.AccountClient;
026import com.nexmo.client.applications.ApplicationClient;
027import com.nexmo.client.auth.AuthMethod;
028import com.nexmo.client.insight.InsightClient;
029import com.nexmo.client.numbers.NumbersClient;
030import com.nexmo.client.sms.SmsClient;
031import com.nexmo.client.sns.SnsClient;
032import com.nexmo.client.verify.VerifyClient;
033import com.nexmo.client.voice.VoiceClient;
034import org.apache.http.client.HttpClient;
035
036/**
037 * Top-level Nexmo API client object.
038 * <p>
039 * Construct an instance of this object with one or more {@link AuthMethod}s (providing all the authentication methods
040 * for the APIs you wish to use), and then call {@link #getVoiceClient()} to obtain a client for the Nexmo Voice API.
041 * <p>
042 * Currently this object only constructs and provides access to {@link VoiceClient}. In the future it will manage
043 * clients for all of the Nexmo APIs.
044 */
045public class NexmoClient {
046    private final AccountClient account;
047    private final ApplicationClient application;
048    private final InsightClient insight;
049    private final NumbersClient numbers;
050    private final SmsClient sms;
051    private final VoiceClient voice;
052    private final VerifyClient verify;
053    private final SnsClient sns;
054
055    private HttpWrapper httpWrapper;
056
057    public NexmoClient(AuthMethod... authMethods) {
058        this.httpWrapper = new HttpWrapper(authMethods);
059
060        this.account = new AccountClient(this.httpWrapper);
061        this.application = new ApplicationClient(this.httpWrapper);
062        this.insight = new InsightClient(this.httpWrapper);
063        this.numbers = new NumbersClient(this.httpWrapper);
064        this.verify = new VerifyClient(this.httpWrapper);
065        this.voice = new VoiceClient(this.httpWrapper);
066        this.sms = new SmsClient(this.httpWrapper);
067        this.sns = new SnsClient(this.httpWrapper);
068    }
069
070    /**
071     * Provide an HttpClient that will be used to make requests to the Nexmo API.
072     * <p>
073     * This can be useful, for example, if you must use an HTTP proxy to make requests.
074     *
075     * @param client A custom-configured HttpClient instance.
076     */
077    public void setHttpClient(HttpClient client) {
078        this.httpWrapper.setHttpClient(client);
079    }
080
081    public AccountClient getAccountClient() {
082        return this.account;
083    }
084
085    public ApplicationClient getApplicationClient() {
086        return this.application;
087    }
088
089    public InsightClient getInsightClient() {
090        return this.insight;
091    }
092
093    public NumbersClient getNumbersClient() {
094        return this.numbers;
095    }
096
097    public SmsClient getSmsClient() {
098        return this.sms;
099    }
100
101    public SnsClient getSnsClient() {
102        return this.sns;
103    }
104
105    public VerifyClient getVerifyClient() {
106        return this.verify;
107    }
108
109    public VoiceClient getVoiceClient() {
110        return this.voice;
111    }
112
113}
