030import com.nexmo.client.voice.endpoints.AbstractMethod;
031import org.apache.commons.logging.Log;
032import org.apache.commons.logging.LogFactory;
033import org.apache.http.HttpResponse;
034import org.apache.http.client.methods.RequestBuilder;
035import org.apache.http.impl.client.BasicResponseHandler;

037import org.w3c.dom.Node;
038import org.w3c.dom.NodeList;
039
040import java.io.IOException;
041import java.io.UnsupportedEncodingException;
042import java.math.BigDecimal;
043import java.util.ArrayList;
044import java.util.List;
045
046public class SendMessageEndpoint extends AbstractMethod<Message, SmsSubmissionResult[]> {
047    private static final Log log = LogFactory.getLog(SmsClient.class);
048
049    private static final Class[] ALLOWED_AUTH_METHODS = new Class[]{TokenAuthMethod.class};
050
051    private static final String DEFAULT_URI = "https://rest.nexmo.com/sms/xml";
052
053    private XmlParser xmlParser = new XmlParser();
054    private String uri = DEFAULT_URI;
055
056    public SendMessageEndpoint(HttpWrapper httpWrapper) {
057        super(httpWrapper);
058    }
059
060    @Override
061    protected Class[] getAcceptableAuthMethods() {
062        return ALLOWED_AUTH_METHODS;
063    }
064
065    @Override
066    public RequestBuilder makeRequest(Message message) throws NexmoClientException, UnsupportedEncodingException {
067        RequestBuilder request = RequestBuilder.post(uri);
068        message.addParams(request);
069        return request;
070    }
071
072    @Override
073    public SmsSubmissionResult[] parseResponse(HttpResponse response) throws IOException {
074        return parseResponse(new BasicResponseHandler().handleResponse(response));
075    }
076
077    protected SmsSubmissionResult[] parseResponse(String response) throws NexmoResponseParseException {
078        // parse the response doc ...
079
080        /*
081            We receive a response from the api that looks like this, parse the document
082            and turn it into an array of SmsSubmissionResult, one object per <message> node
083
084                <mt-submission-response>
085                    <messages count='x'>
086                        <message>
087                            <to>xxx</to>
088                            <messageId>xxx</messageId>
089                            <status>xx</status>
090                            <errorText>ff</errorText>
091                            <clientRef>xxx</clientRef>
092                            <remainingBalance>##.##</remainingBalance>
093                            <messagePrice>##.##</messagePrice>
094                            <reachability status='x' description='xxx' />
095                            <network>23410</network>
096                        </message>
097                    </messages>
098                </mt-submission-response>
099        */
100
101        List<SmsSubmissionResult> results = new ArrayList<>();
102
103        Document doc = xmlParser.parseXml(response);
104
105        NodeList replies = doc.getElementsByTagName("mt-submission-response");
106        for (int i = 0; i < replies.getLength(); i++) {
107            Node reply = replies.item(i);
108            NodeList messageLists = reply.getChildNodes();
109            for (int i2 = 0; i2 < messageLists.getLength(); i2++) {
110                Node messagesNode = messageLists.item(i2);
111                if (messagesNode.getNodeType() != Node.ELEMENT_NODE)
112                    continue;
113                if (messagesNode.getNodeName().equals("messages")) {
114                    NodeList messages = messagesNode.getChildNodes();
115                    for (int i3 = 0; i3 < messages.getLength(); i3++) {
116                        SmsSubmissionResult message = parseMessageXmlNode(messages.item(i3));
117                        if (message != null) {
118                            results.add(message);
119                        }
120                    }
121                }
122            }
123        }
124
125        return results.toArray(new SmsSubmissionResult[results.size()]);
126    }
127
128    private SmsSubmissionResult parseMessageXmlNode(Node messageNode)
129            throws NexmoResponseParseException {
130        if (messageNode.getNodeType() != Node.ELEMENT_NODE)
131            return null;
132
133        int status = -1;
134        String messageId = null;
135        String destination = null;
136        String errorText = null;
137        String clientReference = null;
138        BigDecimal remainingBalance = null;
139        BigDecimal messagePrice = null;
140        SmsSubmissionReachabilityStatus smsSubmissionReachabilityStatus = null;
141        String network = null;
142
143        NodeList nodes = messageNode.getChildNodes();
144        for (int i4 = 0; i4 < nodes.getLength(); i4++) {
145            Node node = nodes.item(i4);
146            if (node.getNodeType() != Node.ELEMENT_NODE)
147                continue;
148            if (node.getNodeName().equals("messageId")) {
149                messageId = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
150            } else if (node.getNodeName().equals("to")) {
151                destination = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
152            } else if (node.getNodeName().equals("status")) {
153                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
154                try {
155                    status = Integer.parseInt(str);
156                } catch (NumberFormatException e) {
157                    log.error("xml parser .. invalid value in <status> node [ " + str + " ] ");
158                    status = SmsSubmissionResult.STATUS_INTERNAL_ERROR;
159                }
160            } else if (node.getNodeName().equals("errorText")) {
161                errorText = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
162            } else if (node.getNodeName().equals("clientRef")) {
163                clientReference = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
164            } else if (node.getNodeName().equals("remainingBalance")) {
165                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
166                try {
167                    if (str != null)
168                        remainingBalance = new BigDecimal(str);
169                } catch (NumberFormatException e) {
170                    log.error("xml parser .. invalid value in <remainingBalance> node [ " + str + " ] ");
171                }
172            } else if (node.getNodeName().equals("messagePrice")) {
173                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
174                try {
175                    if (str != null)
176                        messagePrice = new BigDecimal(str);
177                } catch (NumberFormatException e) {
178                    log.error("xml parser .. invalid value in <messagePrice> node [ " + str + " ] ");
179                }
180            } else if (node.getNodeName().equals("network")) {
181                network = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
182            } else {
183                log.error("xml parser .. unknown node found in status-return, expected [ messageId, to, status, errorText, clientRef, messagePrice, remainingBalance, reachability, network ] -- found [ " + node.getNodeName() + " ] ");
184            }
185        }
186
187        if (status == -1)
188            throw new NexmoResponseParseException("Xml Parser - did not find a <status> node");
189
190        // Is this a temporary error ?
191        boolean temporaryError = (status == SmsSubmissionResult.STATUS_THROTTLED ||
192                status == SmsSubmissionResult.STATUS_INTERNAL_ERROR ||
193                status == SmsSubmissionResult.STATUS_TOO_MANY_BINDS);
194
195        return new SmsSubmissionResult(status,
196                destination,
197                messageId,
198                errorText,
199                clientReference,
200                remainingBalance,
201                messagePrice,
202                temporaryError,
203                smsSubmissionReachabilityStatus,
204                network);
205    }
